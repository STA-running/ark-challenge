## 一、技术选型对比

### 1.1 确认方案：前端网页 + Electron exe（两阶段）

> 砍掉Qt/C++，用最快的技术方案。

```
阶段1 → 前端网页 MVP（验证功能 + 快速上线）
阶段2 → Electron 桌面端 exe（加壳离线化）
```

#### 阶段1：前端网页 + PWA（立即启动）

```
技术栈：Vite + React + TypeScript + Tailwind CSS + shadcn/ui
离线：  PWA Service Worker 缓存
部署：  GitHub Pages / Vercel 免费托管
UI风格：仿明日方舟深色主题（罗德岛绿 + 军事UI感）
```

**理由**：
1. 开发速度最快，社区参考丰富（arkntools就是纯前端）
2. 无需安装，链接分享最方便
3. PWA可离线使用，近似原生体验
4. 与约束方案"分享码"功能天然契合——网页链接 + 分享码

#### 阶段2：Electron 桌面端（前端MVP完成后）

```
技术栈：阶段1代码 + Electron 打包
体积：  ~120MB
优点：  真正离线、可读写本地文件
增量：  只需加 electron-builder 配置，代码复用率95%+
```

### 3.2.1 关卡数据模型（🆕 新增，v3.0 重设计）

> **设计思路**：关卡分类是**两个独立维度**的组合：
> - **类型（type）**：这关"是什么性质"（普通/Boss/超难/超难Boss）
> - **难度模式（difficulty）**：用"什么模式打"（标准/突袭/沙盘推演）
>
> 两者正交组合。例如 "H7-4 星火行动-4" 可以是 "超难Boss关 × 突袭模式"。

```typescript
/**
 * 关卡类型（第一维度：关卡性质）
 *
 * 分类标准：
 * - 普通关：无Boss、无特殊高难机制的常规战斗
 * - Boss关：含命名Boss的关卡（如爱国者、霜星、塔露拉等）
 * - 超难关：非Boss但难度极高的关卡（H关、EX关、部分S关）
 * - 超难Boss关：同时满足Boss+超难的关卡
 */
type StageType = 'normal' | 'boss' | 'hard' | 'hard_boss';
// normal=普通关, boss=Boss关, hard=超难关, hard_boss=超难Boss关

/**
 * 关卡难度模式（第二维度：以什么模式挑战）
 *
 * 说明：
 * - 标准：默认难度
 * - 突袭：更高难度模式。部分关卡称为"磨难"或"险地"，三者等价，统称"突袭"
 * - 沙盘推演：部分关卡特有的额外难度模式
 */
type StageDifficultyMode = 'standard' | 'assault' | 'sandtable';
// standard=标准, assault=突袭(含磨难/险地), sandtable=沙盘推演

/** 关卡所属系列 */
type StageSeries = 'main' | 'side' | 'event' | 'other';
// main=主题曲, side=插曲/别传, event=活动, other=其他

/** 关卡数据 */
interface StageData {
  id: string;                    // 唯一标识，如 "4-10", "H7-4", "DM-EX-6"
  name: string;                  // 关卡名，如 "灯火将熄", "星火行动-4"
  series: StageSeries;           // 所属系列
  chapter: string;               // 章节，如 "第四章 急性衰竭", "生于黑夜"
  type: StageType;               // 关卡类型（第一维度）
  hasSandtable: boolean;         // 是否有沙盘推演模式（默认false）
  // 注意：所有关卡默认含"标准"和"突袭"两种难度模式，无需显式声明
  // 仅沙盘推演需通过此字段标记
}

/**
 * 关卡类型标签映射（用于UI展示）
 */
const STAGE_TYPE_LABELS: Record<StageType, { label: string; icon: string }> = {
  normal:     { label: '普通关',    icon: '⬜' },
  boss:       { label: 'Boss关',    icon: '🟧' },
  hard:       { label: '超难关',    icon: '🟥' },
  hard_boss:  { label: '超难Boss关', icon: '🟪' },
};

/**
 * 难度模式标签映射（用于UI展示）
 */
const DIFFICULTY_MODE_LABELS: Record<StageDifficultyMode, { label: string; note?: string }> = {
  standard:   { label: '标准' },
  assault:    { label: '突袭', note: '部分关卡为磨难/险地模式，等价于突袭' },
  sandtable:  { label: '沙盘推演' },
};

/** 关卡池（索引结构） */
interface StagePool {
  stages: StageData[];
  byType: Record<StageType, StageData[]>;
  bySeries: Record<StageSeries, StageData[]>;
  sandtableStages: StageData[];  // 预索引：有沙盘推演的关卡
```

> 💡 关卡分类是**类型 × 难度模式**两个维度的组合。类型决定"打什么性质的关"，难度模式决定"用什么难度打"。所有关卡默认包含标准和突袭两种模式，沙盘推演通过 `hasSandtable` 标记。

### 3.3 标签合并引擎（v2.0 核心）

```typescript
/**
 * 将多个标签的约束合并为最终约束
 * 
 * 合并规则：
 * - 数值类：取更严格的值（如 maxSquadSize: 4 vs 6 → 取4）
 * - 数组类-白名单：取交集（如 allowedProfessions: [近卫,重装] vs [近卫,医疗] → [近卫]）
 * - 数组类-黑名单：取并集（如 bannedProfessions: [医疗] + [先锋] → [医疗,先锋]）
 * - 布尔类（软约束）：取并集（任一标签开启则开启）
 */
function mergeTagsToConstraints(tags: ChallengeTag[]): {
  hard: HardConstraints;
  soft: SoftConstraints;
  conflicts: TagConflict[];
} {
  const hard = defaultHardConstraints();
  const soft = defaultSoftConstraints();
  const conflicts: TagConflict[] = [];

  for (const tag of tags) {
    // 合并硬约束
    mergeHardConstraints(hard, tag.hardConstraints);
    // 合并软约束
    mergeSoftConstraints(soft, tag.softConstraints);
  }

  // 冲突检测
  conflicts.push(...detectConflicts(tags, hard));

  return { hard, soft, conflicts };
}

/** 数值合并：取更严格（更小）的值 */
function mergeNumber(current: number | null, incoming: number | null): number | null {
  if (current === null) return incoming;
  if (incoming === null) return current;
  return Math.min(current, incoming);  // 上限取更小
}

/** 黑名单合并：取并集 */
function mergeBannedList<T>(current: T[], incoming: T[]): T[] {
  return [...new Set([...current, ...incoming])];
}

/** 冲突检测 */
interface TagConflict {
  tagIds: [string, string];  // 冲突的两个标签
  reason: string;             // 冲突原因
  severity: 'error' | 'warning';  // 冲突等级
}

function detectConflicts(tags: ChallengeTag[], hard: HardConstraints): TagConflict[] {
  const conflicts: TagConflict[] = [];
  
  // 例：同时选了"单职业:近卫"和"无先锋"→ 冲突（近卫不是先锋但逻辑矛盾）
  // 例：同时选了"少人挑战(2人)"和"6星限制(1个)"→ 警告（只能1个6星+1个非6星，很难配）
  
  // 空编队检测
  if (hard.squadSize.maxSquadSize !== null && hard.squadSize.maxSquadSize <= 0) {
    conflicts.push({
      tagIds: tags.map(t => t.id) as [string, string],
      reason: '约束过严，可能无法组建编队',
      severity: 'error'
    });
  }

  return conflicts;
}
```

### 3.4 随机抽奖引擎

```typescript
/** Fisher-Yates 洗牌算法，从标签池中随机抽取N个标签 */
function drawRandomTags(
  pool: TagPool,
  count: number,            // 抽取数量 2~4
  options?: {
    categories?: TagCategory[];  // 限定分类
    maxDifficulty?: number;       // 最高难度
    ensureHardConstraint?: boolean; // 至少包含1个硬约束标签
  }
): ChallengeTag[] {
  let candidates = pool.tags;
  
  // 过滤
  if (options?.categories) {
    candidates = candidates.filter(t => options.categories!.includes(t.category));
  }
  if (options?.maxDifficulty) {
    candidates = candidates.filter(t => t.difficulty <= options.maxDifficulty!);
  }

  // 洗牌
  const shuffled = [...candidates];
  for (let i = shuffled.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
  }

  return shuffled.slice(0, count);
}

/** 单个标签重抽（保持其他标签不变） */
function rerollSingleTag(
  currentTags: ChallengeTag[],
  targetIndex: number,
  pool: TagPool
): ChallengeTag[] {
  const usedIds = new Set(currentTags.map(t => t.id));
  const sameCategory = pool.byCategory[currentTags[targetIndex].category]
    .filter(t => !usedIds.has(t.id) || t.id === currentTags[targetIndex].id);
  
  const alternatives = sameCategory.filter(t => t.id !== currentTags[targetIndex].id);
  if (alternatives.length === 0) return currentTags;
  
  const newTag = alternatives[Math.floor(Math.random() * alternatives.length)];
  const result = [...currentTags];
  result[targetIndex] = newTag;
  return result;
}
```

### 3.4.1 关卡随机抽取引擎（🆕 新增）

```typescript
/** 从关卡池中按难度筛选并随机抽取1个关卡 */
function drawRandomStage(
  pool: StagePool,
  options?: {
    allowedDifficulties?: StageDifficulty[];  // 允许的难度等级
    stageType?: StageType;                     // 限定关卡类型
  }
): StageData | null {
  let candidates = pool.stages;
  
  if (options?.allowedDifficulties && options.allowedDifficulties.length > 0) {
    candidates = candidates.filter(s => options.allowedDifficulties!.includes(s.difficulty));
  }
  if (options?.stageType) {
    candidates = candidates.filter(s => s.stageType === options.stageType);
  }
  
  if (candidates.length === 0) return null;
  
  // Fisher-Yates 随机选取
  const idx = Math.floor(Math.random() * candidates.length);
  return candidates[idx];
}
```

> 💡 关卡抽取引擎和标签抽取引擎使用相同的 Fisher-Yates 洗牌算法，架构统一。

```typescript
  maxRarity: Rarity | null;              // 最高稀有度
  minRarity: Rarity | null;              // 最低稀有度
  maxSixStarCount: number | null;        // 6星数量上限
}

/** 费用约束 */
interface CostConstraints {
  maxTotalCost: number | null;           // 编队总费用上限
  maxSingleCost: number | null;          // 单干员费用上限
  minSingleCost: number | null;          // 单干员费用下限
}

/** 干员黑白名单 */
interface OperatorListConstraints {
  allowedOperators: string[] | null;     // 允许干员白名单(ID列表)
  bannedOperators: string[];             // 禁用干员黑名单(ID列表)
}

/** 位置约束 */
interface PositionConstraints {
  positionRestriction: ('MELEE' | 'RANGED')[] | null; // 限制近战/远程
}

/** 阵营约束（🆕 v3.1 新增） */
interface FactionConstraints {
  sameNationOnly: boolean;             // 仅同国家/地区（编队所有干员 nation 相同，空串忽略）
  sameOrgOnly: boolean;                // 仅同组织/团队（编队所有干员 org 相同，空串忽略）
  allowedNations: string[] | null;      // 允许国家/地区白名单（如 ["炎-龙门","乌萨斯"]）
  bannedNations: string[];             // 禁用国家/地区黑名单
  allowedOrgs: string[] | null;        // 允许组织/团队白名单（如 ["莱茵生命","彩虹小队"]）
  bannedOrgs: string[];                // 禁用组织/团队黑名单
  // 设计说明：
  //   - nation 填充率约90%，org 填充率约10%，两个维度互补
  //   - sameNationOnly/sameOrgOnly 的含义：编队中有 nation/org 的干员必须全部相同，
  //     无 nation/org 的干员不参与约束（否则大量干员无法上场）
  //   - 白名单/黑名单互斥，同职业约束
}

/** 种族约束（🆕 v3.2 新增） */
interface RaceConstraints {
  allowedRaces: string[] | null;       // 允许种族白名单（如 ["萨卡兹","菲林"]）
  bannedRaces: string[];               // 禁用种族黑名单
  sameRaceOnly: boolean;               // 仅同种族（编队所有干员 race 相同，空串忽略）
  rareRaceOnly: number | null;         // 仅稀有种族：只允许该种族总人数≤此值的干员（如 rareRaceOnly=5 表示仅可用人数≤5的稀有种族）
  // 设计说明：
  //   - race 填充率约 98%（仅8名机器人无种族），几乎全覆盖
  //   - 共 44 种种族，人数分布极不均匀：Top3（菲林51/黎博利42/萨卡兹29）共122人，
  //     而19个种族人数≤5（共约40人）
  //   - sameRaceOnly 比 sameNationOnly 更通用（98%覆盖率 vs 90%）
  //   - rareRaceOnly 是独特的约束类型：不限制具体种族，而是限制"种族人数"
  //     例如rareRaceOnly=5 时，只有杜林(5)、龙(4)、德拉克(3)等19个中小种族的干员可用
  //   - 白名单/黑名单互斥，同职业约束
}

/** 阻挡数约束（🆕 v3.3 新增） */
interface BlockConstraints {
  blockTier: 0 | 1 | 2 | 3 | null;  // 阻挡数档位限制：0/1/2/3+；null=无限制
  blockTierMode: 'exact' | 'max';   // exact=必须完全匹配；max=不超过此档位
  maxSameBlockTier: number | null;   // 编队中同一阻挡数档位人数上限，null=无限制
  // 设计说明：
  //   - 面板阻挡数为准，不考虑技能效果（游戏内许多干员开技能变阻挡数）
  //   - exact 模式最严格：blockTier=1 时编队只能选阻挡1的干员（281人）
  //   - max 模式更宽松：blockTier=1 & mode=max 时0和1挡都能用（288人）
  //   - 阻挡数分布：0挡7人（全特种伏击客）/ 1挡281人 / 2挡103人 / 3+挡26人（全重装）
  //   - maxSameBlockTier 用于阻挡多样性挑战："任何挡位不超过N人"
}

/** 合并所有硬约束 */
interface HardConstraints {
  squadSize: SquadSizeConstraints;
  profession: ProfessionConstraints;
  rarity: RarityConstraints;
  cost: CostConstraints;
  operatorList: OperatorListConstraints;
  position: PositionConstraints;
  faction: FactionConstraints;          // 🆕 v3.1
  race: RaceConstraints;                // 🆕 v3.2
  block: BlockConstraints;              // 🆕 v3.3
}

/** ===== 软约束：声明式，靠自律 ===== */

interface SoftConstraints {
  noSkill: boolean;        // 禁用主动技能
  afkOnly: boolean;        // 仅挂机
  noRetreat: boolean;      // 禁止撤退
  noRedeploy: boolean;     // 禁止再部署
  zeroLeak: boolean;       // 零漏击
  maxSteps: number | null; // 操作步数上限（华莱士流）
  customRules: string[];   // 自定义规则文本
}

interface ChallengeScheme {

  stageId: string | null;         // 关联关卡ID（可选）
```

#### 5.1.0.0 增量更新算法（🆕 v3.3 关键设计）

**核心原则**：静态数据（种族、阻挡数）几乎不变，每次更新不重复爬取。动态数据（阵营、新干员）才通过网络获取。

**算法流程**：

```
用户点击"更新"按钮：

① 读取本地缓存的 operators.json（包含所有人的 race + block）
② 从 PRTS Cargo 拉取最新干员列表（1~2次API请求）

③ 对比新旧 charId 集合：
   旧ID集合 = {char_001_12fce, char_002_amiya, ...}
   新ID集合 = {char_001_12fce, char_002_amiya, ..., char_9999_new}

   新增 = 新ID - 旧ID  → 需要爬取完整数据
   不变 = 新ID ∩ 旧ID  → 只更新 nation/org
   移除 = 旧ID - 新ID  → 保留但标记 _removed

④ 仅对"新增"干员爬取 WikiText（通常0~3人，最多1次API请求）
⑤ 合并：老干员保留 race/block，覆盖 nation/org；新干员写入完整数据

      合计：2次Cargo + (0~1)次WikiText = 只需 2~3 次请求 ✅
```

**请求量对比**：

| 场景 | full 模式（开发者） | quick 模式（用户） |
|------|-------------------|-------------------|
| 游戏无更新 | 11次 | **2次** ✅ |
| 新活动出3个干员 | 11次 | **3次** ✅ |
| 大版本更新（假设出50人） | 11次 | **3次** ✅ |

**脚本用法**：

```bash
# 完整构建（开发者首次运行、大版本更新）
python build_operators.py --mode=full

# 快速更新（用户在网页点"更新"按钮时后端执行）
python build_operators.py --mode=quick
# 或指定旧数据文件：
python build_operators.py --mode=quick --input=backup.json
```
- 约 300+ 干员，2次API请求（limit=500+offset分页）即可拉完

#### 5.1.0.2 种族数据说明（🆕 v3.2 新增）

> **数据来源特殊说明**：种族（race）信息**不在 PRTS Cargo 结构化数据中**，而是存在于每个干员页面 wikitext 的 "人员档案set" 模板中，格式为 `|种族=萨卡兹`。
>
> 已通过 PRTS API 批量提取完成（9次 API 调用覆盖 417 名干员），结果保存为 `docs/operator_races.json`，
> 在构建干员数据 JSON 时需与此文件合并。

**提取方式**：

```python
# 核心提取逻辑
# 1. 从 Cargo chara 表获取干员名称列表（417人）
# 2. 通过 MediaWiki action=query&prop=revisions&rvprop=content 批量获取每个干员页面的 wikitext（每次50人，共9次）
# 3. 正则提取：re.search(r'\|\s*种族\s*=\s*([^\n|]+)', wikitext)
```

**覆盖率**：

| 指标 | 数值 |
|------|------|
| 有种族的干员 | 409人（98.1%） |
| 无种族 | 8人（全为机器人单位） |
| 种族种类 | 44种 |

**种族数据分布特征**：

| 类别 | 种族数 | 代表 |

| 大种族（≥10人） | 13种 | 菲林(51)、黎博利(42)、萨卡兹(29)等 |

- race 内部存中文名（PRTS 原值），如 `"萨卡兹"`、`"菲林"`

### 标签黑名单管理（🆕 v3.3）

**功能描述**：用户可以将不喜欢的标签拉入"黑名单"，拉黑后该标签从随机抽取池和标签浏览列表中隐藏。用户也可随时恢复。

**数据结构**：

```typescript
// 黑名单直接存标签ID列表，存在 localStorage 中
// key: 'ark_challenge_tag_blacklist'
// value: JSON 字符串数组，如 '["wallace", "afk_only", "rare_race"]'

interface TagBlacklistManager {
  /** 当前黑名单标签ID集合（用Set去重，快速查） */
  blacklist: Set<string>;

  /** 将标签加入黑名单 */
  add(tagId: string): void;

  /** 将标签移出黑名单 */
  remove(tagId: string): void;

  /** 检查标签是否在黑名单中 */
  isBlacklisted(tagId: string): boolean;

  /** 获取未被拉黑的标签列表 */
  getActiveTags(allTags: ChallengeTag[]): ChallengeTag[];

  /** 获取已被拉黑的标签列表 */
  getBlacklistedTags(allTags: ChallengeTag[]): ChallengeTag[];

  /** 持久化到 localStorage */
  save(): void;

  /** 从 localStorage 恢复 */
  load(): void;
}
```

**影响的范围**：

| 功能 | 黑名单影响 | 说明 |
|------|-----------|------|
| 🎲 随机抽奖 | ✅ 黑名单标签不参与抽取 | 抽奖池 = 全部标签 - 黑名单标签 |
| 🏷️ 手动选标签 | ✅ 黑名单标签默认隐藏 | 标签浏览页面顶部加"显示已拉黑标签"开关 |
| 📋 已有方案 | ❌ 不受影响 | 已保存的方案即使包含已拉黑标签，仍可正常查看和编辑 |

**持久化**：

```typescript
// 保存
localStorage.setItem('ark_challenge_tag_blacklist', JSON.stringify([...this.blacklist]));

// 恢复
const saved = localStorage.getItem('ark_challenge_tag_blacklist');
if (saved) {
  const list = JSON.parse(saved);
  this.blacklist = new Set(list);
}
```

### 编队预览布局（SquadPreview）

**网格规格**：
- 2行 × 6列 CSS Grid，最大容纳12人
- 每个干员卡片固定 92×104px
- 填充顺序：**列优先**，即 (行1,列1) → (行2,列1) → (行1,列2) → (行2,列2) → ...
- 干员数不足时，右边和下边直接留白，**不渲染空位占位元素**

**实现方式**：

```css
.squad-grid {
  display: grid;
  grid-template-rows: repeat(2, auto);
  grid-auto-flow: column;
  gap: 8px;
}
```

`grid-auto-flow: column` 配合 `grid-template-rows: repeat(2, auto)` 天然实现列优先填充。不填充无干员的网格单元，CSS Grid 会自动压缩右侧空白。

**每个卡片的渲染**：

```
┌──────────┐
│     ✕    │  ← 悬停显示删除按钮
│  (头像)  │  ← 40px圆形：优先PRTS头像URL(60px缩略图)，加载失败回退职业图标
│  干员名   │  ← 11px 500
│  费用+阻挡 │  ← 10px muted（如 "💰17 🛡️2"；违规时显示"超限!"，红色）
└──────────┘
```

**头像加载策略**（fallback链）：

```
PRTS 60px缩略图 → 加载失败 → PRTS 职业图标(795B) → 加载失败 → CSS绘制的职业emoji
```

**填充分示例**（以7人为例）：

```
上排：[op1] [op3] [op5] [op7]
下排：[op2] [op4] [op6]
      ↑ 从左到右按列填充，右侧留白
```

### 关卡黑名单管理（🆕 与标签黑名单分区）

**功能描述**：用户可按章节批量拉黑关卡，拉黑后该章节的所有关卡不再出现在随机关卡抽取中。与标签黑名单在 UI 上独立分区、使用不同的 localStorage key，防止混淆。

**为什么按章节而不是按单个关卡**：
- 通关进度是按章节推进的，用户可能主线只打到第六章，第七章还没解锁
- 如果一关一关拉黑，几十个关卡选到手软
- 按章节拉黑，一次操作就能过滤一整章

**数据结构**：

### 5.2 合规校验引擎（硬约束验证）

```typescript
/** 校验单干员是否满足硬约束 */
function validateOperator(
  operator: OperatorData,
  hard: HardConstraints,
  currentSquad: OperatorData[]
): { valid: boolean; reasons: string[] } {
  const reasons: string[] = [];
  // 部署费用：统一使用精二费用（精英化不可逆，不能限制培养进度）
  const cost = operator.deployCost.elite2;

  // B.职业检查（白名单/黑名单互斥）
  if (hard.profession.allowedProfessions && 
      !hard.profession.allowedProfessions.includes(operator.profession)) {
    reasons.push(`职业不在白名单中`);
  }
  if (hard.profession.bannedProfessions.includes(operator.profession)) {
    reasons.push(`${getProfessionName(operator.profession)}职业已被禁用`);
  }

  // B.分支检查
  if (hard.profession.allowedSubProfessions && 
      !hard.profession.allowedSubProfessions.includes(operator.subProfessionId)) {
    reasons.push(`分支不在白名单中`);
  }
  if (hard.profession.bannedSubProfessions.includes(operator.subProfessionId)) {
    reasons.push(`${getSubProfessionName(operator.subProfessionId)}分支已被禁用`);
  }

  // C.稀有度检查
  if (hard.rarity.maxRarity !== null && operator.rarity > hard.rarity.maxRarity) {
    reasons.push(`★${operator.rarity + 1}超出最高稀有度限制★${hard.rarity.maxRarity + 1}`);
  }
  if (hard.rarity.minRarity !== null && operator.rarity < hard.rarity.minRarity) {
    reasons.push(`★${operator.rarity + 1}低于最低稀有度要求★${hard.rarity.minRarity + 1}`);
  }
  // C.6星数量检查（需结合现有编队）
  if (hard.rarity.maxSixStarCount !== null && operator.rarity === 5) {
    const currentSixStars = currentSquad.filter(op => op.rarity === 5).length;
    if (currentSixStars >= hard.rarity.maxSixStarCount) {
      reasons.push(`6星干员数量已达上限(${hard.rarity.maxSixStarCount})`);
    }
  }

  // D.费用检查
  if (hard.cost.maxSingleCost !== null && cost > hard.cost.maxSingleCost) {
    reasons.push(`部署费用${cost}超出单干员上限${hard.cost.maxSingleCost}`);
  }
  if (hard.cost.minSingleCost !== null && cost < hard.cost.minSingleCost) {
    reasons.push(`部署费用${cost}低于下限${hard.cost.minSingleCost}`);
  }

  // E.干员黑白名单
  if (hard.operatorList.allowedOperators && 
      !hard.operatorList.allowedOperators.includes(operator.id)) {
    reasons.push(`干员不在白名单中`);
  }
  if (hard.operatorList.bannedOperators.includes(operator.id)) {
    reasons.push(`该干员已被禁用`);
  }

  // F.位置检查
  if (hard.position.positionRestriction &&
      !hard.position.positionRestriction.includes(operator.position)) {
    reasons.push(`${operator.position === 'MELEE' ? '近战' : '远程'}位不在允许范围`);
  }

  // G.阵营检查（🆕 v3.1）
  if (hard.faction.allowedNations && operator.nation &&
      !hard.faction.allowedNations.includes(operator.nation)) {
    reasons.push(`${operator.nation}不在允许的国家/地区中`);
  }
  if (hard.faction.bannedNations.includes(operator.nation)) {
    reasons.push(`${operator.nation}已被禁用`);
  }
  if (hard.faction.allowedOrgs && operator.org &&
      !hard.faction.allowedOrgs.includes(operator.org)) {
    reasons.push(`${operator.org}不在允许的组织中`);
  }
  if (hard.faction.bannedOrgs.includes(operator.org)) {
    reasons.push(`${operator.org}已被禁用`);
  }

  // H.种族检查（🆕 v3.2）
  if (hard.race.allowedRaces && operator.race &&
      !hard.race.allowedRaces.includes(operator.race)) {
    reasons.push(`${operator.race}不在允许的种族中`);
  }
  if (hard.race.bannedRaces.includes(operator.race)) {
    reasons.push(`种族${operator.race}已被禁用`);
  }
  if (hard.race.rareRaceOnly !== null && operator.race) {
    const raceTotal = getRaceCount(operator.race); // 从干员总库获取种族人数
    if (raceTotal > hard.race.rareRaceOnly) {
      reasons.push(`种族${operator.race}干员数(${raceTotal})超出稀有种族上限(${hard.race.rareRaceOnly})`);
    }
  }

  // I.阻挡数检查（🆕 v3.3）
  if (hard.block.blockTier !== null && operator.block !== undefined) {
    if (hard.block.blockTierMode === 'exact') {
      if (operator.block !== hard.block.blockTier) {
        reasons.push(`阻挡数${operator.block}不匹配要求(${hard.block.blockTier})`);
      }
    } else if (hard.block.blockTierMode === 'max') {
      if (operator.block > hard.block.blockTier) {
        reasons.push(`阻挡数${operator.block}超出上限(${hard.block.blockTier})`);
      }
    }
  }

  return { valid: reasons.length === 0, reasons };
}

/** 校验整个编队是否满足硬约束 */
function validateSquad(
  squad: OperatorData[],
  hard: HardConstraints
): { passed: boolean; violations: string[] } {
  const violations: string[] = [];

  // A.编队人数
  if (hard.squadSize.maxSquadSize !== null && 
      squad.length > hard.squadSize.maxSquadSize) {
    violations.push(`编队人数${squad.length}超出上限${hard.squadSize.maxSquadSize}`);
  }

  // A.同职业人数
  if (hard.squadSize.maxSameProfession !== null) {
    const profCount = countByProfession(squad);
    for (const [prof, count] of Object.entries(profCount)) {
      if (count > hard.squadSize.maxSameProfession) {
        violations.push(`${getProfessionName(prof as Profession)}人数${count}超出上限${hard.squadSize.maxSameProfession}`);
      }
    }
  }

  // D.总费用
  const totalCost = squad.reduce((sum, op) => sum + op.deployCost.elite2, 0);
  if (hard.cost.maxTotalCost !== null && totalCost > hard.cost.maxTotalCost) {
    violations.push(`总部署费用${totalCost}超出上限${hard.cost.maxTotalCost}`);
  }

  // C.6星数量
  if (hard.rarity.maxSixStarCount !== null) {
    const sixCount = squad.filter(op => op.rarity === 5).length;
    if (sixCount > hard.rarity.maxSixStarCount) {
      violations.push(`6星干员${sixCount}个，超出上限${hard.rarity.maxSixStarCount}`);
    }
  }

  // G.同阵营约束（🆕 v3.1）
  if (hard.faction.sameNationOnly) {
    const nations = [...new Set(squad.filter(op => op.nation).map(op => op.nation))];
    if (nations.length > 1) {
      violations.push(`仅同国家/地区：编队含多种归属 ${nations.join('、')}`);
    }
  }
  if (hard.faction.sameOrgOnly) {
    const orgs = [...new Set(squad.filter(op => op.org).map(op => op.org))];
    if (orgs.length > 1) {
      violations.push(`仅同组织/团队：编队含多个组织 ${orgs.join('、')}`);
    }
  }

  // H.同种族约束（🆕 v3.2）
  if (hard.race.sameRaceOnly) {
    const races = [...new Set(squad.filter(op => op.race).map(op => op.race))];
    if (races.length > 1) {
      violations.push(`仅同种族：编队含多种族 ${races.join('、')}`);
    }
  }

  // I.同阻挡数上限（🆕 v3.3）
  if (hard.block.maxSameBlockTier !== null) {
    const blockGroups: Record<number, string[]> = {};
    for (const op of squad) {
      if (op.block === undefined) continue;
      const tier = op.block >= 3 ? 3 : op.block; // 3+统一归为"3"档
      if (!blockGroups[tier]) blockGroups[tier] = [];
      blockGroups[tier].push(op.name);
    }
    for (const [tier, ops] of Object.entries(blockGroups)) {
      if (ops.length > hard.block.maxSameBlockTier) {
        violations.push(`阻挡${tier}的干员(${ops.length}人)超出同阻挡上限(${hard.block.maxSameBlockTier})`);
      }
    }
  }

  // 逐个干员检查
  squad.forEach(op => {
    const result = validateOperator(op, hard, squad.filter(s => s.id !== op.id));
    violations.push(...result.reasons.map(r => `${op.name}: ${r}`));
  });

  return { passed: violations.length === 0, violations };
}
```

### 5.3 分享码实现（🆕 超短位掩码编码）

> **设计原则**：分享码只编码**规则**（标签+关卡+软约束开关），不编码具体干员选了谁。挑战的本质是规则，阵容是每个人自己决定的。

#### 编码格式

```
┌──────────────────────────────────────────────────────────────┐
│  版本(4bit) │ 标签位掩码(26bit) │ 关卡索引(16bit)            │
│  ──────────┼─────────────────┼────────────────────────── │
│  软约束开关(6bit) │ 覆盖标记(1bit) │ [覆盖数据(可选)]          │
└──────────────────────────────────────────────────────────────┘

  固定头部: 4 + 26 + 16 + 6 + 1 = 53 bits ≈ 7 bytes
  Base64URL编码: 7字节 → 10字符
  加前缀 "AKRL:": 共 15字符
  
  含覆盖数据时: +4~10字节 → 总计 17~27字符
```

#### 位字段说明

| 字段 | 位数 | 说明 |
|------|------|------|
| `version` | 4 bit | 版本号，预留兼容（当前=1） |
| `tagMask` | 26 bit | 标签位掩码，bit 0-19 对应标签索引 0-19；bit 20-25 为🆕 v3.2 新增种族标签（萨卡兹之力等6个） |
| `stageIndex` | 16 bit | 关卡在 stages.json 中的索引（0-65535） |
| `softFlags` | 6 bit | 6个软约束开关：noSkill/afkOnly/noRetreat/noRedeploy/zeroLeak/hasCustom |
| `overrideFlag` | 1 bit | 是否有手动覆盖约束值 |
| `overrideData` | 可变 | 仅 overrideFlag=1 时存在，编码覆盖字段 |

#### 实现代码

```typescript
/** 标签ID → 位掩码索引的映射表（在 tags.json 加载时构建） */
let tagIdToIndex: Map<string, number> = new Map();

/** 构建标签索引映射 */
function buildTagIndex(tags: ChallengeTag[]): void {
  tagIdToIndex.clear();
  tags.forEach((tag, idx) => tagIdToIndex.set(tag.id, idx));
}

/** 编码：方案 → 超短分享码 */
function encodeScheme(
  scheme: ChallengeScheme,
  tags: ChallengeTag[],
  stages: StageData[]
): string {
  const view = new DataView(new ArrayBuffer(8)); // 8字节足够
  let offset = 0;
  
  // 1. 版本号 (4 bit)
  view.setUint8(0, 1 << 4); // version=1, 高4位
  offset = 4;
  
  // 2. 标签位掩码 (20 bit)
  let tagMask = 0;
  for (const tagId of scheme.selectedTags) {
    const idx = tagIdToIndex.get(tagId);
    if (idx !== undefined) tagMask |= (1 << idx);
  }
  // 写入20bit到offset位置
  writeBits(view, offset, tagMask, 20);
  offset += 20;
  
  // 3. 关卡索引 (16 bit)
  const stageIdx = stages.findIndex(s => s.id === scheme.stageId);
  writeBits(view, offset, Math.max(0, stageIdx), 16);
  offset += 16;
  
  // 4. 软约束开关 (6 bit)
  let softFlags = 0;
  if (scheme.softConstraints.noSkill)    softFlags |= 0b100000;
  if (scheme.softConstraints.afkOnly)    softFlags |= 0b010000;
  if (scheme.softConstraints.noRetreat)  softFlags |= 0b001000;
  if (scheme.softConstraints.noRedeploy) softFlags |= 0b000100;
  if (scheme.softConstraints.zeroLeak)   softFlags |= 0b000010;
  if (scheme.softConstraints.customRules.length > 0) softFlags |= 0b000001;
  writeBits(view, offset, softFlags, 6);
  offset += 6;
  
  // 5. 覆盖标记 (1 bit)
  writeBits(view, offset, scheme.manuallyOverridden ? 1 : 0, 1);
  offset += 1;
  
  // 6. 覆盖数据（可选，仅当 manuallyOverridden=true）
  // 用精简编码：只编码与标签默认值不同的字段
  // （实现时用 LZ-string 压缩差异数据，通常4-10字节）
  
  // 转为 Base64URL
  const byteLength = Math.ceil(offset / 8);
  const bytes = new Uint8Array(view.buffer, 0, byteLength);
  const base64 = uint8ArrayToBase64URL(bytes);
  
  return `AKRL:${base64}`;
}

/** 解码：超短分享码 → 方案 */
function decodeScheme(
  code: string,
  tags: ChallengeTag[],
  stages: StageData[],
  defaultHardConstraints: () => HardConstraints,
  defaultSoftConstraints: () => SoftConstraints
): ChallengeScheme | null {
  if (!code.startsWith('AKRL:')) return null;
  try {
    const base64 = code.slice(5);
    const bytes = base64URLToUint8Array(base64);
    const view = new DataView(bytes.buffer);
    let offset = 0;
    
    // 1. 版本号
    const version = (view.getUint8(0) >> 4) & 0xF;
    if (version !== 1) return null;
    offset = 4;
    
    // 2. 标签位掩码
    const tagMask = readBits(view, offset, 20);
    offset += 20;
    const selectedTagIds: string[] = [];
    for (let i = 0; i < 20; i++) {
      if (tagMask & (1 << i)) {
        selectedTagIds.push(tags[i].id);
      }
    }
    
    // 3. 关卡索引
    const stageIdx = readBits(view, offset, 16);
    offset += 16;
    const stage = stageIdx < stages.length ? stages[stageIdx] : null;
    
    // 4. 软约束开关
    const softFlags = readBits(view, offset, 6);
    offset += 6;
    const soft = defaultSoftConstraints();
    soft.noSkill    = !!(softFlags & 0b100000);
    soft.afkOnly    = !!(softFlags & 0b010000);
    soft.noRetreat  = !!(softFlags & 0b001000);
    soft.noRedeploy = !!(softFlags & 0b000100);
    soft.zeroLeak   = !!(softFlags & 0b000010);
    // customRules 从覆盖数据段恢复（如有）
    
    // 5. 覆盖标记
    const hasOverride = readBits(view, offset, 1) === 1;
    
    // 6. 通过标签合并生成硬约束
    const selectedTags = selectedTagIds.map(id => tags.find(t => t.id === id)!).filter(Boolean);
    const { hard } = mergeTagsToConstraints(selectedTags);
    
    // 7. 如果有覆盖数据，合并覆盖
    if (hasOverride) {
      // 解码覆盖差异数据并应用（实现时补充）
    }
    
    return {
      id: crypto.randomUUID(),
      name: selectedTags.map(t => t.name).join('+'),
      description: '',
      stageId: stage?.id ?? null,
      stageName: stage?.name ?? null,
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
      selectedTags: selectedTagIds,
      hardConstraints: hard,
      softConstraints: soft,
      selectedOperators: [],
      manuallyOverridden: hasOverride,
    };
  } catch {
    return null;
  }
}
```

### 5.5 一键数据更新服务（🆕 安全防刷设计）

> **设计目标**：一键从 PRTS Wiki Cargo API 拉取最新数据，但绝不给 Wiki 服务器添麻烦。

#### 5.5.1 为什么不用担心？

先算一笔账——一次完整更新到底产生多少请求：

| 数据 | Cargo API 请求 | 说明 |
|------|----------------|------|
| 干员数据 | 2个请求 | limit=500，约312个干员，1次拉不完的用offset翻页 |
| 关卡数据 | 1个请求 | 关卡数量少，1次搞定 |
| **合计** | **3个请求** | 加2秒间隔 ≈ 6秒完成 |

对比正常用户浏览 PRTS Wiki：
- 打开1个干员页面 = 10+个HTTP请求（HTML+CSS+JS+图片+API）
- 浏览5个干员 = 50+个请求

**结论：我们3个API请求 ≈ 用户浏览0.3个页面**，完全不用担心搞崩。但仍然加防护，防手贱和恶意使用。

#### 5.5.2 三层防刷机制

```typescript
/** 数据更新状态 */
interface DataUpdateState {
  lastUpdateAt: string | null;      // 上次更新时间 ISO 8601
  operatorCount: number;            // 当前干员数量
  stageCount: number;               // 当前关卡数量
  isUpdating: boolean;              // 是否正在更新中
  updateProgress: UpdateProgress;   // 更新进度
}

interface UpdateProgress {
  step: string;        // 当前步骤描述
  current: number;      // 当前步骤
  total: number;        // 总步骤数
  status: 'idle' | 'fetching' | 'processing' | 'saving' | 'done' | 'error';
  error?: string;       // 错误信息
}

/** 冷却期配置 */
const COOLDOWN_MS = 24 * 60 * 60 * 1000;  // 24小时
const REQUEST_INTERVAL_MS = 2000;           // 请求间隔2秒

/** 检查是否在冷却期内 */
function canUpdate(state: DataUpdateState): { allowed: boolean; nextUpdateAt: string | null } {
  if (!state.lastUpdateAt) return { allowed: true, nextUpdateAt: null };
  
  const lastUpdate = new Date(state.lastUpdateAt).getTime();
  const nextAllowed = lastUpdate + COOLDOWN_MS;
  const now = Date.now();
  
  if (now >= nextAllowed) {
    return { allowed: true, nextUpdateAt: null };
  }
  
  return {
    allowed: false,
    nextUpdateAt: new Date(nextAllowed).toISOString()
  };
}
```

#### 5.5.3 更新流程实现

```typescript
/** 一键更新入口 */
async function updateAllData(
  state: DataUpdateState,
  onProgress: (progress: UpdateProgress) => void
): Promise<DataUpdateResult> {
  // 0. 检查冷却期
  const { allowed, nextUpdateAt } = canUpdate(state);
  if (!allowed) {
    return {
      success: false,
      reason: `冷却中，下次可更新: ${nextUpdateAt}`,
    };
  }
  
  // 0.5 防止重复触发
  if (state.isUpdating) {
    return { success: false, reason: '正在更新中，请勿重复点击' };
  }

  // 1. 自动备份旧数据（失败自动回滚用）
  const backup = {
    operators: localStorage.getItem('arknights-operators-backup') ?? 
               await fetch('/data/operators.json').then(r => r.text()),
    stages: localStorage.getItem('arknights-stages-backup') ?? 
            await fetch('/data/stages.json').then(r => r.text()),
  };
  localStorage.setItem('arknights-operators-backup', backup.operators);
  localStorage.setItem('arknights-stages-backup', backup.stages);

  try {
    // 2. 拉取干员数据（2个请求）
    onProgress({ step: '干员数据 (1/2)', current: 1, total: 4, status: 'fetching' });
    const operators1 = await fetchFromPRTS('chara,chara_data', 0);
    await sleep(REQUEST_INTERVAL_MS);  // ⏱️ 2秒间隔

    onProgress({ step: '干员数据 (2/2)', current: 2, total: 4, status: 'fetching' });
    const operators2 = await fetchFromPRTS('chara,chara_data', 500);
    
    // 3. 数据清洗 + 合并
    onProgress({ step: '数据清洗', current: 2, total: 4, status: 'processing' });
    const allOperators = [...operators1, ...operators2]
      .map(cleanOperatorData)          // 清洗HTML标签、解析数字
      .filter(op => op !== null);       // 过滤无效数据
    
    await sleep(REQUEST_INTERVAL_MS);  // ⏱️ 2秒间隔

    // 4. 拉取关卡数据（1个请求）
    onProgress({ step: '关卡数据', current: 3, total: 4, status: 'fetching' });
    const stages = await fetchStagesFromPRTS();

    // 5. 保存到 localStorage（覆盖运行时数据）
    onProgress({ step: '保存数据', current: 4, total: 4, status: 'saving' });
    localStorage.setItem('arknights-operators', JSON.stringify(allOperators));
    localStorage.setItem('arknights-stages', JSON.stringify(stages));
    localStorage.setItem('arknights-last-update', new Date().toISOString());

    onProgress({ step: '完成', current: 4, total: 4, status: 'done' });
    
    return {
      success: true,
      operatorCount: allOperators.length,
      stageCount: stages.length,
    };
    
  } catch (error) {
    // 💥 出错了！自动回滚到备份数据
    console.error('数据更新失败，正在回滚:', error);
    localStorage.setItem('arknights-operators', backup.operators);
    localStorage.setItem('arknights-stages', backup.stages);
    
    onProgress({ step: '更新失败', current: 0, total: 4, status: 'error', error: String(error) });
    
    return {
      success: false,
      reason: `更新失败: ${error}，已自动回滚到旧数据`,
    };
  }
}

/** PRTS Wiki Cargo API 请求封装 */
async function fetchFromPRTS(
  tables: string,
  offset: number
): Promise<any[]> {
  const params = new URLSearchParams({
    action: 'cargoquery',
    format: 'json',
    limit: '500',
    offset: String(offset),
    tables: tables,
    fields: 'chara.cn,chara.charId,chara.rarity,chara.profession,'
           + 'chara.subProfession,chara.position,'
           + 'chara.nation,chara.team,'                        // 🆕 v3.1 阵营字段
           + 'chara_data.cost,chara_data.block,chara_data.reDeploy',
    join_on: 'chara._pageName=chara_data._pageName',
  });
  
  const response = await fetch(`https://prts.wiki/api.php?${params}`);
  if (!response.ok) {
    throw new Error(`PRTS API 请求失败: HTTP ${response.status}`);
  }
  
  const data = await response.json();
  return data.cargoquery?.map((item: any) => item.title) ?? [];
}
```

#### 5.5.5 关卡数据维护策略：手动 JSON（v3.0 修正）

> **为什么不爬 PRTS 关卡**：即使有 24h 冷却+2s 间隔，若工具被数万玩家使用，
> 合计请求量仍可能冲击 PRTS 服务器。关卡数据量远小于干员数据，手动维护成本可控。
> 
> **设计原则**：关卡分类是**自定义体系**（普通/Boss/超难/超难Boss × 标准/突袭/沙盘推演），
> PRTS 无法提供此数据。因此手动维护 `stages.json` 是最合理的选择。

##### 维护方式

| 策略 | 说明 |
|------|------|
| **数据来源** | 手动编辑 `public/data/stages.json` |
| **更新时机** | 游戏版本更新后，由工具维护者手动添加新关卡 |
| **更新方式** | 打开 JSON 文件，在 `stages` 数组末尾追加新对象 |
| **难度定级** | 由维护者判断并填写 `type` 字段（normal/boss/hard/hard_boss） |
| **沙盘推演** | 个别关卡设 `hasSandtable: true` |

```
       "name": "绝境",

       "chapter": "众生行记",
       "type": "hard",        ← 维护者自行判断
       "hasSandtable": false

  → 保存文件，发布新版本
```

> 💡 JSON 格式简单（6个字段），复制一个已有条目改改就行。新增关卡只能追加到数组末尾，
> 不能调整已有条目顺序（关卡索引用于分享码编码）。

##### 一键更新流程（仅干员数据，v3.0 修正）

一键更新**仅拉取干员数据**，不涉及关卡数据：

```typescript
/** 一键更新入口（v3.0：仅干员数据，关卡手动维护） */
async function updateAllData(
  state: DataUpdateState,
  onProgress: (progress: UpdateProgress) => void
): Promise<DataUpdateResult> {
  // 0. 检查冷却期 & 防重复触发（同 §5.5.3）
  // 1. 自动备份旧数据（同 §5.5.3）

  try {
    // 2. 拉取干员数据（2个 Cargo API 请求，同 §5.5.3）
    onProgress({ step: '拉取干员数据 (1/2)', current: 1, total: 3, status: 'fetching' });
    const operators1 = await fetchFromPRTS('chara', 0, 500);
    await sleep(REQUEST_INTERVAL_MS);
    
    onProgress({ step: '拉取干员数据 (2/2)', current: 2, total: 3, status: 'fetching' });
    const operators2 = await fetchFromPRTS('chara', 500, 500);

    // 3. 数据清洗 + 合并
    onProgress({ step: '数据清洗', current: 3, total: 3, status: 'processing' });
    const allOperators = [...operators1, ...operators2]
      .map(cleanOperatorData)
      .filter(op => op !== null);

    // 4. 保存干员数据
    localStorage.setItem('arknights-operators', JSON.stringify(allOperators));

    onProgress({ step: '完成', current: 3, total: 3, status: 'done' });

    return {
      success: true,
      operatorCount: allOperators.length,
    };
  } catch (error) {
    // 出错自动回滚（同 §5.5.3）
  }
}
```

|------|---------|--------|------|--------|
| 干员数据 | Cargo API | 2 | 2s | ~4s |
| 关卡数据 | **不请求** | 0 | - | 0s |
| **合计** | | **2** | | **~4s** |

## 十一、已确认决策汇总

| 决策项 | 结论 |
|--------|------|
| 项目定位 | **自限挑战规则器** — 帮玩家制定自律规则，不是游戏修改器 |
| 约束体系 | **硬约束**（工具可验证）+ **软约束**（声明式自律）双轨 |
| 🆕 标签驱动 | **标签 = 自限玩法的封装**，标签名含规则+指标值（如"编队≤4""总费≤80""★≤3"） |
| 🆕 标签详细说明 | `detail` 字段提供消除歧义的完整解释，标签卡片独立区域展示 |
| 🆕 标签数据外置 | 标签数据存为 `public/data/tags.json`，改文件即改前端，无需动代码 |
| 🆕 关卡系统 | 关卡5级难度分级，随机抽取+手动选择，和标签同模式 |
| 🆕 关卡数据外置 | 关卡数据存为 `public/data/stages.json`，改文件即改前端，无需动代码 |
| 🆕 超短分享码 | 位掩码编码，只编码标签+关卡+软约束开关，13~25字符，移除lz-string依赖 |
| 🆕 一键数据更新 | 增量更新机制：24h冷却期+2s请求间隔+自动备份回滚，仅2次Cargo API请求/次（无新干员时），最多3次（新干员需爬WikiText），远低于正常浏览 |
| 🆕 增量更新原理 | 通过 charId 差异对比自动发现新干员，静态数据（种族/阻挡数）本地保留，不重复爬取 |
| 🆕 标签黑名单 | localStorage 持久化，不喜欢的标签可拉黑（不在随机池和列表中出现），可随时恢复。不影响已有方案 |
| 🆕 干员头像 | 外链PRTS 60px缩略图，不打包进安装包。构建时通过 MediaWiki API 批量获取URL，100%覆盖 |
| 🆕 关卡黑名单 | 按章节批量拉黑，与标签黑名单分区展示、独立 localStorage key。拉黑章节不参与随机关卡抽取 |
| 🆕 开发者模式 | 点击标题版本号3次切换开发模式，显示数据更新按钮。用户版本隐藏此按钮，防止误操作 |
| 🆕 关卡数据更新策略 | 自动发现+手动定级：从PRTS关卡一览页面解析HTML发现新关卡，用户在UI中分配5级难度后写入stages.json（不依赖Cargo API） |
| 🆕 种族数据来源 | **PRTS 页面 wikitext**（"人员档案set"模板的 `|种族=xxx`），**非 Cargo API 结构化数据**。已通过 PRTS API 批量提取完成，保存为 `docs/operator_races.json`，构建时合并到干员数据 |
| 🆕 阻挡数数据来源 | **PRTS 页面 wikitext**（"人员档案set"模板的 `|阻挡数=1`），与种族同批次提取，**100%覆盖**（417/417人）。以面板数据为准，不考虑技能效果 |
| 🆕 种族约束 | 6个种族标签（萨卡兹之力/菲林成群/黎博利飞行队/稀有种族/种族大熔炉/禁常见种族），占位掩码从20bit扩至26bit |
| 🆕 阻挡数约束 | 4个阻挡数标签（零阻挡/一阻挡/二阻挡/三阻挡），面板数据为准无视技能效果，100%覆盖，占位掩码从26bit扩至30bit |
| 精英化不可逆 | ⚠️ **不提供"限制精英化等级"选项**；费用验证统一以精二为准；elite0/elite1仅作展示 |
| 交付形态 | **两阶段**：前端网页 → Electron exe（砍掉Qt） |
| 干员头像 | 不需要，用职业图标+名字 |
| 数据更新 | PRTS Wiki Cargo API 爬取（已验证可用） |
| 账号系统 | 纯本地，不需要 |
| UI风格 | 仿游戏深色主题 |
